Housekeeping¶

In [110]:
# Drop the spurious data column

df.drop(df.columns[-1], axis=1, inplace=True)


# Rename the datetime column

df.rename(columns={'LOGGING:':'Timestamp'}, inplace=True)


# Remove trailing semicolon in datetime values

df.Timestamp = df.Timestamp.apply(lambda x: x.rstrip(':'))


# Convert all columns to numeric datatype, and drop rows with non-numeric data

df = df.apply(pd.to_numeric, errors='coerce').dropna(axis=0)


# Convert 'Timestamp' column to datetime type, then make it the index

df.Timestamp = pd.to_datetime(df.Timestamp, unit='s')
df.set_index('Timestamp', inplace=True)


# Drop seconds information from the datetime index

df.index = df.index.map(lambda x: x.replace(second=0))


# Drop rows with duplicate index values

df = df[~df.index.duplicated(keep='last')]

# Convert timestamp to local time.

my_timezone = pytz.timezone('US/Eastern')
df.index = df.index.tz_localize(pytz.utc).tz_convert(my_timezone)


# Convert mV --> V and mA --> A

df[['VS','IS','VB','IB','IC','VM','TH']] = df[['VS','IS','VB','IB','IC','VM','TH']]/1000.0


# Convert from Celcius to Farenheit

df[['IT','ET']] = df[['IT','ET']].apply(lambda degC: degC/10.0*1.8+32 )
